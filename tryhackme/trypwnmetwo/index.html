    shell.sendlineafter(b">>", b"1")

    shell.sendafter(b":\n", data)

    shell.sendlineafter(b">>", b"2")
    shell.sendlineafter(b":\n", str(index).encode())

    shell.sendlineafter(b">>", b"3")
    shell.sendlineafter(b":\n", str(index).encode())

    shell.sendlineafter(b">>", b"4")
    shell.sendlineafter(b":\n", str(index).encode())

def win(index):
    shell.sendlineafter(b">>", b"5")
    shell.sendlineafter(b":\n", str(index).encode())
